class Fc:
    def __init__(self, n_in, n_out, relu=True, seed=0):
        self.n_in=n_in
        self.n_out=n_out
        self.relu=relu
        # w,b 
        # ReLu
        # ReLU=TrueHeFalseXavier
        # xforwardBP
        np.random.seed(seed) #  seed 
        if relu:
            self.w=np.random.normal(0, math.sqrt(2.0/n_in), (n_out, n_in)) # He
            self.relu0=None # ReLUBP
        else:
            self.w=np.random.normal(0, math.sqrt(1.0/n_in), (n_out, n_in)) # Xavier
        self.b=np.zeros((n_out,1))
        self.x=np.zeros((n_in,1))
        #(Back-Propagation)
        self.dEdx=np.zeros((n_in,1)) # dE/dx BP
        self.dEdw=np.zeros((n_out,n_in)) # dE/dw 
        self.dEdb=np.zeros((n_out,1)) # dE/db 
        self.count=0
        # MomentamSDG
        self.mdw=np.zeros((n_out,n_in))
        self.mdb=np.zeros((n_out,1))

    # forward  __call__  
    def __call__(self,x): 
        self.x = x
        self.y = np.dot(self.w, x) + self.b
        # ReLU relu0BP
        if self.relu:
            self.relu0= self.y<=0
            self.y[self.relu0]=0
        return self.y

    # backward dE/dy dE/dx dE/dw, dE/db 
    def backward(self, dEdy):
          # 
          if self.relu:
              dEdy[self.relu0]=0
          dydx = np.transpose(self.w)  # dWx/dx= np.transpose(W) 
          dydw = np.transpose(self.x)  # dWx/dW= np.transpose(x) 
          # dydb = np.ones((self.n_out,1))   #  db/db = np.ones((n_out,1))
          # dEdw, dEdb dEdx ()backward
          self.dEdx = np.dot(dydx, dEdy)
          self.dEdw += np.dot(dEdy, dydw)
          self.dEdb += dEdy
          self.count +=1  # 
          return self.dEdx

    # 
    def clear_grad(self):
        self.dEdw=0
        self.dEdb=0
        self.count=0

    # count
    def update(self, lr=0.001):
        self.w -= self.dEdw/self.count * lr
        self.b -= self.dEdb/self.count * lr
        self.clear_grad()
    
    # 
    def updatem(self, lr=0.001, mu=0.9):
        self.mdw = mu*self.mdw - self.dEdw/self.count * lr
        self.mdb = mu*self.mdb - self.dEdb/self.count * lr
        self.w += self.mdw
        self.b += self.mdb
        self.clear_grad()

        fc2.update(lr)
        fc3.update(lr)
        # epoch

            y0=np.reshape(Y_pred,[-1])
            fig2.plot(x0,y0,"b")

                fig2.cla()
